fields @timestamp, @message, @requestId, @logStream
| sort @timestamp desc
| limit 10000
| filter @message like /ERROR/
  or @message like /Error/
  or @message like /Exception/
  or @message like /exception/
  or @message like /Traceback/
  or @message like /failed/i
  or @message like /FAILED/
  or @level = "ERROR"
  or @level = "FATAL"
